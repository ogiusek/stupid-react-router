## Install

```bash
npm install --save stupid-react-router
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'stupid-react-router'
import 'stupid-react-router/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
